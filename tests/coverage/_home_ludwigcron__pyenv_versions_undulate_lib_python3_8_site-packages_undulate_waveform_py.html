109def _prune_json(filepath: str): 

110 """ 

111 assume the file exists and: 

112 - remove comments 

113 - replace single quotes by double quotes around string 

114 - remove final extra comma 

115 - parse numbers 

116 """ 

117 ans = {} 

118 with open(filepath, "r+") as fp: 

119 content = " ".join( 

120 [ 

121 line[: line.find("//")] if line.find("//") >= 0 else line 

122 for line in fp.readlines() 

123 ] 

124 ) 

125 # add double quotes around strings 

126 content = re.sub(r"([{,]?)\s*(\w+)\s*:", r'\1 "\2":', content, flags=re.M) 

127 # replace single quotes with double quotes 

128 content = re.sub( 

129 r"'([\w\s\<\-\~\|\>,*\_\.:\[\]\(\)]*)\s*'", r'"\1"', content, flags=re.M 

130 ) 

131 # remove final extra comma of arrays definition 

132 content = re.sub(r"(,\s*\])", r"]", content, flags=re.M) 

133 # change hex numbers to int 

134 content = re.sub("0'?([xbhdXBHD])([0-9ABCDEF]+)", _number_convert, content, flags=re.M) 

135 tmp = json.loads(content) 

136 for k, v in tmp.items(): 

137 if k == "signal": 

138 for _, sig in enumerate(v): 

139 n, wave = ( 

140 _parse_wavelane(sig) if isinstance(sig, dict) else _parse_group(sig) 

141 ) 

142 if n in ans.keys(): 

143 print("Signal %s is duplicated" % n) 

144 while n in ans:

147 else: 

148 ans[k] = v

152def parse(filepath: str) -> (bool, object): 

153 """ 

154 parse the input file into a compatible dict for processing 

155 """ 

156 err, ans = False, {} 

157 # file existence 

158 err = filepath is None or not os.path.exists(filepath) 

159 if err: 

160 log_Error(ERROR_MSG["FILE_NOT_FOUND"] % filepath) 

161 return (err, None) 

162 _, ext = os.path.splitext(filepath) 

163 # supported extension 

164 err = not any([ext in cat for cat in SUPPORTED_FORMAT.values()]) 

165 if err: 

166 log_Error(ERROR_MSG["UNSUPPORTED_FORMAT"]) 

167 return (err, None) 

168 # call appropriate parser 

169 if ext in SUPPORTED_FORMAT["json"]: 

170 ans.update(_prune_json(filepath)) 

171 elif ext in SUPPORTED_FORMAT["yaml"]: 

172 try: 

173 import yaml 

174 

175 with open(filepath, "r+") as fp: 

176 ans = yaml.load(fp, Loader=yaml.Loader) 

177 except ImportError: 

178 log_Error(ERROR_MSG["YAML_IMPORT"]) 

179 else: 

180 try: 

181 import toml

183 with open(filepath, "r+") as fp: 

184 ans = toml.load(fp) 

185 except ImportError: 

186 log_Error(ERROR_MSG["TOML_IMPORT"]) 

187 return (err, ans if not err else None)

190def register_to_wavelane(obj: dict) -> object: 

191 """ 

192 convert a register definition as a wavelane 

193 """ 

194 reg = undulate.Register() 

195 # name of the register 

196 reg.name = [name for name in obj.keys() if name not in ["config", "head", "foot"]][-1] 

197 for field in obj.get(reg.name, []): 

198 reg.push_field(field) 

199 # default value from wavedrom format 

200 if reg.name == "reg": 

201 reg.name = "" 

202 reg.config = obj.get("config", {}) 

203 return reg.to_wavelane()
